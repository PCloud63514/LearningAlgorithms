SELECT CART_ID
FROM (
    SELECT CART_ID,
    SUM(CASE WHEN(NAME = 'Milk') THEN 1 ELSE 0 END) AS BUY_MILK, # 같은 것을 여러개 구매했을 수 있으므로, SUM
    SUM(CASE WHEN(NAME = 'Yogurt') THEN 1 ELSE 0 END) AS BUY_YOGURT
    FROM CART_PRODUCTS
    GROUP BY CART_ID
) cp
WHERE 0 < cp.BUY_MILK and 0 < cp.BUY_YOGURT
ORDER BY cp.CART_ID ASC
